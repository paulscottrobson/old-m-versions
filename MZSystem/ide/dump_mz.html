------------------------------------------------------------------------------------------------
output a row of 8 bytes
------------------------------------------------------------------------------------------------

dump.bytes private
ab>r
8 for
con.space swap a>r c@ con.print.byte r>a 1+ swap
next
r>ab
;

************************************************************************************************
dump memory starting at a ; on exit a is the next block
************************************************************************************************

dump.mem
b>r
8 for
b>r swap con.print.word con.space 58 con.emit r>a
dump.bytes
8 + swap
con.cr
next
swap
r>b
;
crunch
